### subscribeProvider

```ts
import { onMount } from 'svelte'
import { modal } from '$lib/appkit'

onMount(() => {
  if (!modal) return

  const unsubscribe = modal.subscribeProvider((newState) => {
    console.log(newState)
  })

  return unsubscribe
})
```
